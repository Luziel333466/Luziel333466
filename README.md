```python
def criar_musica():
    # Verso 1
    verso1 = "Nossos caminhos se perderam,\nEntre sombras de um amor que já se foi.\nPalavras não ditas, silêncio cortante,\nNada mais resta, nosso fim ecoou."

    # Refrão
    refrao = "Nosso amor acabou e não volta mais,\nComo folhas ao vento, se desfez.\nA despedida pesa no peito,\nResta aceitar, seguir o que se fez."

    # Verso 2
    verso2 = "Promessas quebradas, sonhos desfeitos,\nNo labirinto da desilusão.\nCada adeus ecoa como um lamento,\nNo coração, só resta solidão."

    # Refrão
    refrao_repetido = "Nosso amor acabou e não volta mais,\nComo folhas ao vento, se desfez.\nA despedida pesa no peito,\nResta aceitar, seguir o que se fez."

    # Ponte
    ponte = "O que foi, ficou no passado,\nMemórias que um dia foram lar.\nMas agora, é hora de partir,\nCaminhar sem olhar para trás."

    # Refrão final
    refrao_final = "Nosso amor acabou e não volta mais,\nComo folhas ao vento, se desfez.\nA despedida pesa no peito,\nResta aceitar, seguir o que se fez."

    # Unir todas as partes
    musica = f"{verso1}\n\n{refrao}\n\n{verso2}\n\n{refrao_repetido}\n\n{ponte}\n\n{refrao_final}"

    return musica

# Chamada da função para obter a música
musica_final = criar_musica()

# Imprimir a música
print(musica_final)
```
